### 14. Örnek Kapsamlı Program

```otağ
# Değişken tanımlama
x'ı tamsayı olarak tanımla
isim = "Ali"
toplam_değer'i tamsayı olarak tanımla

# Fonksiyon
fonksiyon selamla! isim: "Ali"
    söyle "Merhaba " + isim

# Koşul
eğer x > 3 ise
    söyle "x büyük"
aksi halde
    söyle "x küçük veya eşit"

# Döngü
x < 10 iken tekrarla
    söyle x
    x'i 1 artır

# Standart kütüphane fonksiyonları
dene
    dosyaya_yaz! dosya: "notlar.txt", içerik: "Merhaba Dünya" eğer hatalıysa devam et
eğer hata varsa
    söyle "Dosya yazma hatası"
aksi halde
    söyle "Dosya başarıyla yazıldı"
her daim
    söyle "İşlem tamamlandı"

alis_listesi = yeni liste
alis_listesi.ekle! değer: 42
telefon_defteri = yeni sözlük
telefon_defteri.ekle! anahtar: "Ali", değer: "555-1234"
```
